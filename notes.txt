import java.util.Map;
import java.util.stream.Collectors;
import java.util.stream.Stream;

public class StringToMapExample {
    public static void main(String[] args) {
        String input = "site1:url1|site2:url2|site3:url3";

        // Convert the string to a Map
        Map<String, String> map = Stream.of(input.split("\\|")) // Split by pipe
                .map(entry -> entry.split(":")) // Split each entry by colon
                .collect(Collectors.toMap(
                        parts -> parts[0], // Key is the first part
                        parts -> parts[1]  // Value is the second part
                ));

        // Print the map
        System.out.println(map);
    }
}
